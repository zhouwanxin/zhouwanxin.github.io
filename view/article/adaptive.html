              vue+vant移动端适配

              hei小于 1 分钟约 248 字
              此页内容
              • 先安装amfe-flexible和postcss-pxtorem：
              • 在main.js导入amfe-flexible
              • 在vue.config.js配置如下：
              • 如果是vue/cli @5.x等高版本脚手架时，在vue.config.js可以尝试如下配置：
              • 如果有需要将px转换为rem，则使用

              # 先安装amfe-flexible和postcss-pxtorem：

              npm install amfe-flexible --save
npm install postcss-pxtorem@5.1.1

              # 在main.js导入amfe-flexible

              import 'amfe-flexible';

              # 在vue.config.js配置如下：

              module.exports = {
//...其他配置
 css: {
        loaderOptions: {
            postcss: {
                plugins: [
                    require('postcss-pxtorem')({
                        rootValue({ file }) {
                            return file.indexOf('vant') !== -1 ? 37.5 : 75;
                        },
                        propList: ['*']
                    })
                ]
            }
        }
    }
}
// rootValue根据设计稿宽度除以10进行设置，这边假设设计稿为750，即rootValue设为75；
// file.indexOf('vant') !== -1 vant保持375的设计稿
// propList是设置需要转换的属性，这边*为所有都进行转换。

              # 如果是vue/cli @5.x等高版本脚手架时，在vue.config.js可以尝试如下配置：

              module.exports = {
//...其他配置
 css: {
        loaderOptions: {
            postcss: {
                postcssOptions: {
                    plugins: [
                        require('postcss-pxtorem')({
                            rootValue({ file }) {
                                return file.indexOf('vant') !== -1 ? 37.5 : 75;
                            },
                            propList: ['*']
                        })
                    ]
                }
            }
        }
    }
}

              # 如果有需要将px转换为rem，则使用

              px2rem(px) {
    if (/%/gi.test(px)) {
        // 有百分号%，特殊处理，表述pc是一个有百分号的数，比如：90%
        return px;
    } else {
        return parseFloat(px) / 75 + "rem"; // 这里的75，和rootValue值对应
    }
},
